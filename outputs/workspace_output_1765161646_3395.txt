import React, { useEffect, useRef, useState } from 'react';
import { AvatarState, AvatarMood, VisemeData } from '../hooks/useJeanAvatarState';
import { useJeanAvatar } from '../hooks/useJeanAvatar';

interface JeanAvatarProps {
  config?: {
    model_url?: string;
    voice_id?: string;
    size?: 'small' | 'medium' | 'large';
    show_controls?: boolean;
    auto_start?: boolean;
    theme?: 'light' | 'dark';
  };
  className?: string;
  onStateChange?: (state: AvatarState) => void;
  onMoodChange?: (mood: AvatarMood) => void;
  onVisemeUpdate?: (viseme: VisemeData | null) => void;
}

const JeanAvatar: React.FC<JeanAvatarProps> = ({
  config = {},
  className = '',
  onStateChange,
  onMoodChange,
  onVisemeUpdate,
}) => {
  const canvasRef = useRef<HTMLCanvasElement>(null);
  const animationRef = useRef<number>();
  const [isVideoPlaying, setIsVideoPlaying] = useState(false);
  const [currentVideoFrame, setCurrentVideoFrame] = useState<string | null>(null);

  const avatarConfig = {
    model_url: '/models/jean_avatar.glb',
    voice_id: 'jean_default',
    size: 'medium',
    show_controls: true,
    auto_start: false,
    theme: 'dark',
    ...config,
  };

  const {
    state: avatarState,
    config: internalConfig,
    actions,
    isReady,
  } = useJeanAvatar({
    model_url: avatarConfig.model_url,
    voice_id: avatarConfig.voice_id,
    animation_speed: 1.0,
    auto_blink: true,
    responsiveness: 0.8,
    mood_sensitivity: 0.6,
  });

  // Size configurations
  const sizeConfig = {
    small: { width: 120, height: 120, fontSize: 12 },
    medium: { width: 200, height: 200, fontSize: 14 },
    large: { width: 300, height: 300, fontSize: 16 },
  };

  const currentSize = sizeConfig[avatarConfig.size];

  // Handle state changes
  useEffect(() => {
    onStateChange?.(avatarState.state);
  }, [avatarState.state, onStateChange]);

  useEffect(() => {
    onMoodChange?.(avatarState.mood);
  }, [avatarState.mood, onMoodChange]);

  useEffect(() => {
    onVisemeUpdate?.(avatarState.current_viseme);
  }, [avatarState.current_viseme, onVisemeUpdate]);

  // Canvas animation loop
  useEffect(() => {
    const canvas = canvasRef.current;
    if (!canvas || !isReady) return;

    const ctx = canvas.getContext('2d');
    if (!ctx) return;

    const animate = () => {
      // Clear canvas
      ctx.clearRect(0, 0, canvas.width, canvas.height);

      // Draw avatar based on state and viseme
      drawAvatar(ctx, canvas.width, canvas.height);

      animationRef.current = requestAnimationFrame(animate);
    };

    animate();

    return () => {
      if (animationRef.current) {
        cancelAnimationFrame(animationRef.current);
      }
    };
  }, [avatarState, isReady]);

  // Draw avatar on canvas
  const drawAvatar = (ctx: CanvasRenderingContext2D, width: number, height: number) => {
    const centerX = width / 2;
    const centerY = height / 2;
    const radius = Math.min(width, height) * 0.35;

    // Background
    ctx.fillStyle = avatarConfig.theme === 'dark' ? '#1a1a1a' : '#f0f0f0';
    ctx.fillRect(0, 0, width, height);

    // Avatar body (circle with mood-based color)
    const bodyGradient = ctx.createRadialGradient(centerX, centerY, 0, centerX, centerY, radius);
    
    switch (avatarState.mood) {
      case 'happy':
        bodyGradient.addColorStop(0, '#4ade80');
        bodyGradient.addColorStop(1, '#22c55e');
        break;
      case 'concerned':
        bodyGradient.addColorStop(0, '#fb923c');
        bodyGradient.addColorStop(1, '#f97316');
        break;
      case 'excited':
        bodyGradient.addColorStop(0, '#fbbf24');
        bodyGradient.addColorStop(1, '#f59e0b');
        break;
      case 'thoughtful':
        bodyGradient.addColorStop(0, '#60a5fa');
        bodyGradient.addColorStop(1, '#3b82f6');
        break;
      case 'confused':
        bodyGradient.addColorStop(0, '#c084fc');
        bodyGradient.addColorStop(1, '#a855f7');
        break;
      default:
        bodyGradient.addColorStop(0, '#9ca3af');
        bodyGradient.addColorStop(1, '#6b7280');
    }

    ctx.fillStyle = bodyGradient;
    ctx.beginPath();
    ctx.arc(centerX, centerY, radius, 0, Math.PI * 2);
    ctx.fill();

    // Draw eyes based on state
    drawEyes(ctx, centerX, centerY, radius);

    // Draw mouth based on viseme or mood
    if (avatarState.current_viseme) {
      drawMouthFromViseme(ctx, centerX, centerY, radius, avatarState.current_viseme);
    } else {
      drawMoodBasedMouth(ctx, centerX, centerY, radius, avatarState.mood);
    }

    // Draw state indicator
    drawStateIndicator(ctx, width, height, avatarState.state);

    // Draw connection status
    drawConnectionStatus(ctx, width, height, avatarState.connection_status);
  };

  const drawEyes = (ctx: CanvasRenderingContext2D, centerX: number, centerY: number, radius: number) => {
    const eyeRadius = radius * 0.08;
    const eyeY = centerY - radius * 0.15;
    const eyeSpacing = radius * 0.3;

    // Left eye
    ctx.fillStyle = avatarConfig.theme === 'dark' ? '#ffffff' : '#000000';
    ctx.beginPath();
    ctx.arc(centerX - eyeSpacing, eyeY, eyeRadius, 0, Math.PI * 2);
    ctx.fill();

    // Right eye
    ctx.beginPath();
    ctx.arc(centerX + eyeSpacing, eyeY, eyeRadius, 0, Math.PI * 2);
    ctx.fill();

    // Eye shine
    ctx.fillStyle = 'rgba(255, 255, 255, 0.6)';
    ctx.beginPath();
    ctx.arc(centerX - eyeSpacing + eyeRadius * 0.3, eyeY - eyeRadius * 0.3, eyeRadius * 0.3, 0, Math.PI * 2);
    ctx.fill();

    ctx.beginPath();
    ctx.arc(centerX + eyeSpacing + eyeRadius * 0.3, eyeY - eyeRadius * 0.3, eyeRadius * 0.3, 0, Math.PI * 2);
    ctx.fill();

    // Blinking animation for listening state
    if (avatarState.state === 'listening' && Math.random() < 0.1) {
      ctx.fillStyle = avatarConfig.theme === 'dark' ? '#1a1a1a' : '#f0f0f0';
      ctx.fillRect(centerX - eyeSpacing - eyeRadius, eyeY - eyeRadius, eyeRadius * 2, eyeRadius * 2);
      ctx.fillRect(centerX + eyeSpacing - eyeRadius, eyeY - eyeRadius, eyeRadius * 2, eyeRadius * 2);
    }
  };

  const drawMoodBasedMouth = (ctx: CanvasRenderingContext2D, centerX: number, centerY: number, radius: number, mood: AvatarMood) => {
    const mouthY = centerY + radius * 0.25;
    const mouthWidth = radius * 0.4;

    ctx.strokeStyle = avatarConfig.theme === 'dark' ? '#ffffff' : '#000000';
    ctx.lineWidth = 3;
    ctx.lineCap = 'round';

    switch (mood) {
      case 'happy':
        ctx.beginPath();
        ctx.arc(centerX, mouthY - radius * 0.1, mouthWidth, 0.2 * Math.PI, 0.8 * Math.PI);
        ctx.stroke();
        break;
      case 'concerned':
        ctx.beginPath();
        ctx.arc(centerX, mouthY + radius * 0.1, mouthWidth, 1.2 * Math.PI, 1.8 * Math.PI);
        ctx.stroke();
        break;
      case 'excited':
        ctx.fillStyle = avatarConfig.theme === 'dark' ? '#ffffff' : '#000000';
        ctx.beginPath();
        ctx.ellipse(centerX, mouthY, mouthWidth * 0.8, mouthWidth * 1.2, 0, 0, Math.PI * 2);
        ctx.fill();
        break;
      case 'thoughtful':
      case 'confused':
        ctx.beginPath();
        ctx.moveTo(centerX - mouthWidth, mouthY);
        ctx.lineTo(centerX + mouthWidth, mouthY);
        ctx.stroke();
        break;
      default:
        ctx.beginPath();
        ctx.ellipse(centerX, mouthY, mouthWidth * 0.3, mouthWidth * 0.1, 0, 0, Math.PI * 2);
        ctx.stroke();
    }
  };

  const drawMouthFromViseme = (ctx: CanvasRenderingContext2D, centerX: number, centerY: number, radius: number, viseme: VisemeData) => {
    const mouthY = centerY + radius * 0.25;
    const blendShapes = viseme.blend_shapes;
    
    ctx.fillStyle = avatarConfig.theme === 'dark' ? '#ffffff' : '#000000';

    // Apply viseme-based mouth shape
    if (blendShapes.jawOpen && blendShapes.jawOpen > 0.1) {
      const mouthHeight = radius * 0.3 * blendShapes.jawOpen;
      const mouthWidth = radius * 0.4;
      
      ctx.beginPath();
      ctx.ellipse(centerX, mouthY + mouthHeight/2, mouthWidth, mouthHeight, 0, 0, Math.PI * 2);
      ctx.fill();
    }

    // Handle specific mouth shapes
    if (blendShapes.lipsRound && blendShapes.lipsRound > 0.5) {
      ctx.beginPath();
      ctx.arc(centerX, mouthY, radius * 0.2, 0, Math.PI * 2);
      ctx.fill();
    }

    if (blendShapes.lipsPressed && blendShapes.lipsPressed > 0.5) {
      ctx.fillRect(centerX - radius * 0.2, mouthY - 2, radius * 0.4, 4);
    }
  };

  const drawStateIndicator = (ctx: CanvasRenderingContext2D, width: number, height: number, state: AvatarState) => {
    const indicatorSize = 8;
    const x = width - indicatorSize - 10;
    const y = 10;

    switch (state) {
      case 'listening':
        ctx.fillStyle = '#3b82f6';
        break;
      case 'speaking':
        ctx.fillStyle = '#10b981';
        break;
      case 'processing':
        ctx.fillStyle = '#f59e0b';
        break;
      case 'error':
        ctx.fillStyle = '#ef4444';
        break;
      default:
        ctx.fillStyle = '#6b7280';
    }

    ctx.beginPath();
    ctx.arc(x, y, indicatorSize / 2, 0, Math.PI * 2);
    ctx.fill();

    // Pulsing animation for active states
    if (state !== 'idle') {
      ctx.strokeStyle = ctx.fillStyle;
      ctx.globalAlpha = 0.3;
      ctx.lineWidth = 2;
      ctx.beginPath();
      ctx.arc(x, y, indicatorSize / 2 + 3, 0, Math.PI * 2);
      ctx.stroke();
      ctx.globalAlpha = 1;
    }
  };

  const drawConnectionStatus = (ctx: CanvasRenderingContext2D, width: number, height: number, status: 'connected' | 'connecting' | 'disconnected') => {
    const x = 10;
    const y = 10;
    const size = 6;

    switch (status) {
      case 'connected':
        ctx.fillStyle = '#10b981';
        break;
      case 'connecting':
        ctx.fillStyle = '#f59e0b';
        break;
      default:
        ctx.fillStyle = '#ef4444';
    }

    ctx.beginPath();
    ctx.arc(x, y, size / 2, 0, Math.PI * 2);
    ctx.fill();
  };

  const handleStartListening = () => {
    actions.startListening();
  };

  const handleStopListening = () => {
    actions.stopListening();
  };

  const handleSpeak = async () => {
    try {
      await actions.startSpeaking('Hello! I am Jean, your AI assistant. How can I help you today?', {
        emotion: 'friendly',
        show_visemes: true,
      });
    } catch (error) {
      console.error('Speech error:', error);
    }
  };

  const handleStopSpeaking = () => {
    actions.stopSpeaking();
  };

  const handleMoodChange = (mood: AvatarMood) => {
    actions.setMood(mood);
  };

  const getAvatarAnimationClass = () => {
    const stateClass = `avatar-${avatarState.state}`;
    const moodClass = `mood-${avatarState.mood}`;
    return `${stateClass} ${moodClass}`;
  };

  return (
    <div 
      className={`jean-avatar ${getAvatarAnimationClass()} ${className}`}
      style={{
        width: currentSize.width,
        height: currentSize.height,
        position: 'relative',
        fontFamily: 'system-ui, -apple-system, sans-serif',
      }}
    >
      {/* Main avatar canvas */}
      <canvas
        ref={canvasRef}
        width={currentSize.width}
        height={currentSize.height}
        style={{
          width: '100%',
          height: '100%',
          borderRadius: '50%',
          boxShadow: avatarConfig.theme === 'dark' 
            ? '0 4px 20px rgba(0, 0, 0, 0.5)' 
            : '0 4px 20px rgba(0, 0, 0, 0.1)',
        }}
      />

      {/* Status overlay */}
      {avatarState.error_message && (
        <div
          style={{
            position: 'absolute',
            bottom: -40,
            left: 0,
            right: 0,
            textAlign: 'center',
            fontSize: currentSize.fontSize - 2,
            color: '#ef4444',
            fontWeight: 500,
          }}
        >
          {avatarState.error_message}
        </div>
      )}

      {/* Controls */}
      {avatarConfig.show_controls && (
        <div
          style={{
            position: 'absolute',
            bottom: -70,
            left: 0,
            right: 0,
            display: 'flex',
            gap: 4,
            justifyContent: 'center',
            flexWrap: 'wrap',
          }}
        >
          <button
            onClick={avatarState.state === 'listening' ? handleStopListening : handleStartListening}
            disabled={avatarState.state === 'speaking'}
            style={{
              padding: '4px 8px',
              fontSize: currentSize.fontSize - 2,
              borderRadius: 4,
              border: '1px solid #d1d5db',
              backgroundColor: avatarState.state === 'listening' ? '#3b82f6' : '#ffffff',
              color: avatarState.state === 'listening' ? '#ffffff' : '#000000',
              cursor: avatarState.state === 'speaking' ? 'not-allowed' : 'pointer',
              opacity: avatarState.state === 'speaking' ? 0.5 : 1,
            }}
          >
            {avatarState.state === 'listening' ? 'ğŸ›‘' : 'ğŸ¤'}
          </button>

          <button
            onClick={avatarState.state === 'speaking' ? handleStopSpeaking : handleSpeak}
            disabled={avatarState.state === 'listening'}
            style={{
              padding: '4px 8px',
              fontSize: currentSize.fontSize - 2,
              borderRadius: 4,
              border: '1px solid #d1d5db',
              backgroundColor: avatarState.state === 'speaking' ? '#10b981' : '#ffffff',
              color: avatarState.state === 'speaking' ? '#ffffff' : '#000000',
              cursor: avatarState.state === 'listening' ? 'not-allowed' : 'pointer',
              opacity: avatarState.state === 'listening' ? 0.5 : 1,
            }}
          >
            {avatarState.state === 'speaking' ? 'ğŸ”‡' : 'ğŸ”Š'}
          </button>

          {/* Mood selector */}
          <select
            value={avatarState.mood}
            onChange={(e) => handleMoodChange(e.target.value as AvatarMood)}
            style={{
              padding: '2px 4px',
              fontSize: currentSize.fontSize - 2,
              borderRadius: 4,
              border: '1px solid #d1d5db',
              backgroundColor: '#ffffff',
            }}
          >
            <option value="neutral">ğŸ˜</option>
            <option value="happy">ğŸ˜Š</option>
            <option value="concerned">ğŸ˜Ÿ</option>
            <option value="excited">ğŸ¤©</option>
            <option value="thoughtful">ğŸ¤”</option>
            <option value="confused">ğŸ˜•</option>
          </select>
        </div>
      )}

      {/* Video frame overlay (for Wav2Lip integration) */}
      {currentVideoFrame && (
        <img
          src={currentVideoFrame}
          alt="Avatar video"
          style={{
            position: 'absolute',
            top: 0,
            left: 0,
            width: '100%',
            height: '100%',
            borderRadius: '50%',
            objectFit: 'cover',
            pointerEvents: 'none',
          }}
        />
      )}

      {/* Battery indicator */}
      <div
        style={{
          position: 'absolute',
          top: 10,
          left: 10,
          fontSize: currentSize.fontSize - 4,
          color: avatarState.battery_level > 20 ? '#10b981' : '#ef4444',
        }}
      >
        ğŸ”‹ {Math.round(avatarState.battery_level)}%
      </div>

      {/* Ready indicator */}
      {!isReady && (
        <div
          style={{
            position: 'absolute',
            top: '50%',
            left: '50%',
            transform: 'translate(-50%, -50%)',
            fontSize: currentSize.fontSize,
            color: avatarConfig.theme === 'dark' ? '#ffffff' : '#000000',
            textAlign: 'center',
          }}
        >
          Loading...
        </div>
      )}
    </div>
  );
};

export default JeanAvatar;
